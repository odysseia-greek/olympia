Queries

metrics

Response

Returns a Metrics

Arguments
Name Description
timeSpan - String
order - String

Example

Query
query Metrics(
  $timeSpan: String,
  $order: String
) {
  metrics(
    timeSpan: $timeSpan,
    order: $order
  ) {
    cpuUnits
    grouped {
      cpuHumanReadable
      cpuRaw
      memoryHumanReadable
      memoryRaw
      name
    }
    memoryUnits
    nodes {
      cpuHumanReadable
      cpuPercentage
      cpuPercentageHumanReadable
      cpuRaw
      memoryHumanReadable
      memoryPercentage
      memoryPercentageHumanReadable
      memoryRaw
      nodeName
    }
    pods {
      cpuHumanReadable
      cpuRaw
      memoryHumanReadable
      memoryRaw
      name
    }
    timeEnded
    timeStamps
    timeStarted
  }
}
Variables
{
  "timeSpan": "abc123",
  "order": "abc123"
}
Response
{
  "data": {
    "metrics": {
      "cpuUnits": "abc123",
      "grouped": [Grouped],
      "memoryUnits": "abc123",
      "nodes": [Node],
      "pods": [Pod],
      "timeEnded": "abc123",
      "timeStamps": ["abc123"],
      "timeStarted": "xyz789"
    }
  }
}
Queries

traces

Description

Get traces

Response

Returns [ParentTrace]

Arguments
Name Description
input - TraceQueryInput

Example

Query
query Traces($input: TraceQueryInput) {
  traces(input: $input) {
    isActive
    items {
      ... on DatabaseSpan {
        hits
        itemType
        namespace
        parentSpanID
        podName
        query
        spanID
        timestamp
        took
      }
      ... on Span {
        action
        itemType
        namespace
        parentSpanID
        podName
        requestBody
        spanID
        status
        timestamp
        took
      }
      ... on Trace {
        host
        itemType
        method
        metrics {
          ...tracingMetricsFragment
        }
        namespace
        parentSpanID
        podName
        spanID
        timestamp
        url
      }
      ... on StartTrace {
        host
        itemType
        method
        metrics {
          ...tracingMetricsFragment
        }
        namespace
        operation
        parentSpanID
        podName
        remoteAddress
        rootQuery
        spanID
        timestamp
        url
      }
      ... on CloseTrace {
        itemType
        metrics {
          ...tracingMetricsFragment
        }
        namespace
        parentSpanID
        podName
        responseBody
        timestamp
      }
    }
    responseCode
    timeEnded
    timeStarted
    totalTime
    traceID
  }
}
Variables
{"input": TraceQueryInput}
Response
{
  "data": {
    "traces": [
      {
        "isActive": true,
        "items": [DatabaseSpan],
        "responseCode": 987,
        "timeEnded": "abc123",
        "timeStarted": "abc123",
        "totalTime": 987,
        "traceID": "xyz789"
      }
    ]
  }
}
