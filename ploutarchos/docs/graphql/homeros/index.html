Queries

answer

Response

Returns an AnswerUnion

Arguments
Name Description
answer - String
comprehensive - Boolean
dialogue - [DialogueInput]
theme - String
set - String
quizType - String
quizWord - String

Example

Query
query Answer(
  $answer: String,
  $comprehensive: Boolean,
  $dialogue: [DialogueInput],
  $theme: String,
  $set: String,
  $quizType: String,
  $quizWord: String
) {
  answer(
    answer: $answer,
    comprehensive: $comprehensive,
    dialogue: $dialogue,
    theme: $theme,
    set: $set,
    quizType: $quizType,
    quizWord: $quizWord
  ) {
    ... on ComprehensiveResponse {
      correct
      foundInText {
        rhemai {
          ...RhemaFragment
        }
      }
      progress {
        averageAccuracy
        timesCorrect
        timesIncorrect
      }
      quizWord
      similarWords {
        dutch
        english
        greek
        linkedWord
        original
      }
    }
    ... on DialogueAnswer {
      answer {
        greek
        place
        speaker
        translation
      }
      input {
        greek
        place
        speaker
        translation
      }
      percentage
      wronglyPlaced {
        correctPlace
        greek
        place
        speaker
        translation
      }
    }
  }
}
Variables
{
  "answer": "xyz789",
  "comprehensive": false,
  "dialogue": [DialogueInput],
  "theme": "abc123",
  "set": "xyz789",
  "quizType": "abc123",
  "quizWord": "xyz789"
}
Response
{"data": {"answer": ComprehensiveResponse}}

quiz

Response

Returns a QuizResponseUnion

Arguments
Name Description
quizType - String
theme - String
set - String

Example

Query
query Quiz(
  $quizType: String,
  $theme: String,
  $set: String
) {
  quiz(
    quizType: $quizType,
    theme: $theme,
    set: $set
  ) {
    ... on QuizResponse {
      options {
        audioUrl
        imageUrl
        option
      }
      quizItem
    }
    ... on DialogueQuiz {
      content {
        greek
        place
        speaker
        translation
      }
      dialogue {
        introduction
        linkToPerseus
        section
        speakers {
          ...SpeakerFragment
        }
      }
      quizMetadata {
        language
      }
      quizType
      set
      theme
    }
  }
}
Variables
{
  "quizType": "abc123",
  "theme": "abc123",
  "set": "abc123"
}
Response
{"data": {"quiz": QuizResponse}}

text

Description

Check the text given

Response

Returns a Text

Arguments
Name Description
sentenceId - String
author - String
answer - String

Example

Query
query Text(
  $sentenceId: String,
  $author: String,
  $answer: String
) {
  text(
    sentenceId: $sentenceId,
    author: $author,
    answer: $answer
  ) {
    input
    levenshtein
    matches {
      index
      word
    }
    mistakes {
      index
      nonMatches {
        index
        levenshtein
        match
        percentage
      }
      word
    }
    quiz
    splitAnswer {
      word
    }
    splitQuiz {
      word
    }
  }
}

  "sentenceId": "abc123",
  "author": "abc123",
  "answer": "xyz789"
}
Response
{
  "data": {
    "text": {
      "input": "xyz789",
      "levenshtein": "xyz789",
      "matches": [matches],
      "mistakes": [mistakes],
      "quiz": "xyz789",
      "splitAnswer": [word],
      "splitQuiz": [word]
    }
  }
